        (clear-screen)
        (draw-filled-circle (add 220 x)
                            (add 170 y)
                            20

      }

      update { ?x ?y ?a
        (draw x y r)
        (update (multiply 100 (cosine a))
                (multiply 100 (sine a))
                (add a 0.1))
      }]

  (update 0 0 0))
