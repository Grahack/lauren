draw: { ?x ?y
  radius: 20

  clear-screen()
  draw-filled-circle(add(220 x)
                     add(200 y)
                     radius
                     "blue")
}

angle: 0

forever({
  orbit-radius: 100
  angle-step: 0.05
  angle: add(angle angle-step)

  draw(multiply(orbit-radius cosine(angle))
       multiply(orbit-radius sine(angle)))
})
