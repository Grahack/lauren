(name [draw { ?x ?y
        (clear-screen)
        (draw-filled-circle (add 220 x)
                            (add 200 y)
                            20
                            "blue")
      }

      orbit-radius 100

      update { ?x ?y ?current-angle
        (draw x y)

        (update (multiply orbit-radius (cosine current-angle))
                (multiply orbit-radius (sine current-angle))
                (add current-angle 0.1))
      }]

  (update 0 0 0))
